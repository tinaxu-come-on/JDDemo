# 克隆项目
git clone https://gitee.com/tina715/jd.git

# 进入项目目录
cd jd

# 安装依赖
npm install

# 建议不要直接使用 cnpm 安装依赖，会有各种诡异的 bug。可以通过如下操作解决 npm 下载速度慢的问题
npm install --registry=https://registry.npm.taobao.org

# 启动服务
npm start

浏览器访问 http://localhost:3000/jd/home
